# Use a pure Lua mapgen instead of the default engine mapgen. When enabled the
# engine is set to "singlenode" and all terrain is generated entirely in Lua
# using 2D noise. This is not the default.
mapgentest_lua_mapgen (Pure Lua mapgen) bool false

# Enable transparency for base (solid) nodes (mapgen_solid, block_loaded). The fallback and edge nodes are always opaque.
mapgentest_transparent_base_nodes (Transparent base/solid nodes) bool true

# Enable transparency for water nodes (mapgen_water). The fallback and edge nodes are always opaque.
mapgentest_transparent_water_nodes (Transparent water nodes) bool true

# Enable the on_block_loaded callback that marks loaded mapblocks with the block_loaded node.
mapgentest_block_loaded_callback (block_loaded callback) bool false

# Enable the LBM that replaces the base mapgen node (mapgen_solid) with the
# block_loaded (green) node each time a mapblock containing it is loaded.
mapgentest_lbm (LBM: replace mapgen_solid with block_loaded) bool false

# Enable the forest biome with log and leaves nodes and small/medium/tall tree decorations.
mapgentest_forest_biome (Forest biome) bool false

# Draw edge markers at mapchunk boundary faces during on_generated to visualize
# chunk boundaries. Applies to both the engine mapgen and the pure Lua mapgen.
mapgentest_helper_grid (Helper grid) bool true
